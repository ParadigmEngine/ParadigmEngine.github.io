18  static std::vector<psl::vec3> generate_tangents(const std::vector<psl::vec3>& positions,
19  const std::vector<psl::vec2>& uvs,
20  const std::vector<uint32_t>& indices)
21  {
22  std::vector<psl::vec3> tangents;
23  tangents.resize(positions.size());
24 
25  for(size_t i = 0; i < indices.size(); i += 3)
26  {
27  // Shortcuts for vertices
28  const auto& v0 = positions[indices[i + 0]];
29  const auto& v1 = positions[indices[i + 1]];
30  const auto& v2 = positions[indices[i + 2]];
31 
32  const auto& u0 = uvs[indices[i + 0]];
33  const auto& u1 = uvs[indices[i + 1]];
34  const auto& u2 = uvs[indices[i + 2]];
35 
36  psl::vec3 edge1 = v1 - v0;
37  psl::vec3 edge2 = v2 - v0;
38  psl::vec2 deltaUV1 = u1 - u0;
39  psl::vec2 deltaUV2 = u2 - u0;
40 
41  float r = 1.0f / (deltaUV1[0] * deltaUV2[1] - deltaUV1[1] * deltaUV2[0]);
42  psl::vec3 tangent = ((edge1 * deltaUV2[1] - edge2 * deltaUV1[1]) * r);
43  tangent = psl::math::normalize(tangent);
44  tangents[indices[i + 0]] = tangent;
45  tangents[indices[i + 1]] = tangent;
46  tangents[indices[i + 2]] = tangent;
47  /*
48  psl::vec3 bitangent = psl::math::normalize((edge2 * deltaUV1[0] - edge1 * deltaUV2[0])*r);
49  v0.m_bitangent = bitangent;
50  v1.m_bitangent = bitangent;
51  v2.m_bitangent = bitangent;
52  */
53  }
54  return tangents;
55  }

424  static core::resource::handle<core::data::geometry_t>
425  create_spherified_cube(core::resource::cache_t& cache, psl::vec3 scale = psl::vec3::one, uint32_t divisions = 4)
426  {
427  scale *= 0.5f;
428  const float step = 1.0f / float(divisions);
429  const psl::vec3 step3(step, step, step);
430 
431  static const psl::vec3 origins[6] = {psl::vec3(-1.0, -1.0, -1.0),
432  psl::vec3(1.0, -1.0, -1.0),
433  psl::vec3(1.0, -1.0, 1.0),
434  psl::vec3(-1.0, -1.0, 1.0),
435  psl::vec3(-1.0, 1.0, -1.0),
436  psl::vec3(-1.0, -1.0, 1.0)};
437  static const psl::vec3 rights[6] = {psl::vec3(2.0, 0.0, 0.0),
438  psl::vec3(0.0, 0.0, 2.0),
439  psl::vec3(-2.0, 0.0, 0.0),
440  psl::vec3(0.0, 0.0, -2.0),
441  psl::vec3(2.0, 0.0, 0.0),
442  psl::vec3(2.0, 0.0, 0.0)};
443  static const psl::vec3 ups[6] = {psl::vec3(0.0, 2.0, 0.0),
444  psl::vec3(0.0, 2.0, 0.0),
445  psl::vec3(0.0, 2.0, 0.0),
446  psl::vec3(0.0, 2.0, 0.0),
447  psl::vec3(0.0, 0.0, 2.0),
448  psl::vec3(0.0, 0.0, -2.0)};
449 
450  std::vector<psl::vec3> vertices;
451 
452  for(uint32_t face = 0; face < 6; ++face)
453  {
454  const psl::vec3 origin = origins[face];
455  const psl::vec3 right = rights[face];
456  const psl::vec3 up = ups[face];
457  for(uint32_t j = 0; j < divisions + 1; ++j)
458  {
459  const psl::vec3 j3((float)j, (float)j, (float)j);
460  for(uint32_t i = 0; i < divisions + 1; ++i)
461  {
462  const psl::vec3 i3((float)i, (float)i, (float)i);
463  const psl::vec3 p = origin + step3 * (i3 * right + j3 * up);
464  const psl::vec3 p2 = p * p;
465  const psl::vec3 n(p[0] * psl::math::sqrt(1.0f - 0.5f * (p2[1] + p2[2]) + p2[1] * p2[2] / 3.0f),
466  p[1] * psl::math::sqrt(1.0f - 0.5f * (p2[2] + p2[0]) + p2[2] * p2[0] / 3.0f),
467  p[2] * psl::math::sqrt(1.0f - 0.5f * (p2[0] + p2[1]) + p2[0] * p2[1] / 3.0f));
468  vertices.emplace_back(n);
469  }
470  }
471  }
472 
473  std::vector<uint32_t> triangles;
474  const uint32_t k = divisions + 1;
475  for(uint32_t face = 0; face < 6; ++face)
476  {
477  for(uint32_t j = 0; j < divisions; ++j)
478  {
479  for(uint32_t i = 0; i < divisions; ++i)
480  {
481  const uint32_t a = (face * k + j) * k + i;
482  const uint32_t b = (face * k + j) * k + i + 1;
483  const uint32_t c = (face * k + j + 1) * k + i;
484  const uint32_t d = (face * k + j + 1) * k + i + 1;
485  triangles.push_back(a);
486  triangles.push_back(b);
487  triangles.push_back(d);
488  triangles.push_back(c);
489  triangles.push_back(a);
490  triangles.push_back(d);
491  }
492  }
493  }

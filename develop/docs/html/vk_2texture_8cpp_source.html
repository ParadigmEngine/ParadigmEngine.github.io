135 void texture_t::create_2D(void* data)
136 {
137  m_MipLevels = m_Meta->mip_levels();
138  vk::FormatProperties formatProperties;
139  if(m_Meta->format() == gfx::format_t::undefined)
140  {
141  LOG_ERROR("Undefined format property in: ", utility::to_string(m_Meta->ID()));
142  }
143  m_Context->physical_device().getFormatProperties(to_vk(m_Meta->format()), &formatProperties);
144 
145  auto stagingBuffer = m_StagingBuffer;
146  vk::BufferImageCopy bufferCopyRegion;
147  if(data != nullptr)
148  {
149  auto size = m_Meta->width() * m_Meta->height();
150  if(!m_StagingBuffer)
151  {
152  core::ivk::log->warn(
153  "missing a staging buffer in ivk::texture_t, will create one dynamically, but this is inefficient");
154  auto tempBuffer = m_Cache.create<core::data::buffer_t>(
155  core::gfx::memory_usage::transfer_source,
156  core::gfx::memory_property::host_visible | core::gfx::memory_property::host_coherent,
157  memory::region {size + 1024, 4, new memory::default_allocator(false)});
158  stagingBuffer = m_Cache.create<ivk::buffer_t>(m_Context, tempBuffer);
159  }
160  if(!stagingBuffer)
161  {
162  core::ivk::log->error("could not create a staging buffer in ivk::texture_t");
163  return;
164  }
165  memory::segment segment;
166  if(auto segmentOpt = stagingBuffer->reserve((vk::DeviceSize)size); segmentOpt)
167  {
168  segment = segmentOpt.value();
169  gfx::commit_instruction instr;
170  instr.segment = segment;
171  instr.size = (vk::DeviceSize)size;
172  instr.source = (std::uintptr_t)data;
173  if(!stagingBuffer->commit({instr}))
174  {
175  core::ivk::log->error("could not commit an ivk::texture_t in a staging buffer");
176  }
177  }
178  else
179  {
180  core::ivk::log->error("could not allocate a segment in the staging buffer for an ivk::texture_t");
181  return;
182  }
183 
184  bufferCopyRegion.imageSubresource.aspectMask = vk::ImageAspectFlagBits::eColor;
185  bufferCopyRegion.imageSubresource.mipLevel = 0;
186  bufferCopyRegion.imageSubresource.baseArrayLayer = 0;
187  bufferCopyRegion.imageSubresource.layerCount = m_Meta->layers();
188  bufferCopyRegion.imageExtent.width = m_Meta->width();
189  bufferCopyRegion.imageExtent.height = m_Meta->height();
190  bufferCopyRegion.imageExtent.depth = 1;
191  bufferCopyRegion.bufferOffset = 0;
192  }
193 
194  // Create optimal tiled target image
195  vk::ImageCreateInfo imageCreateInfo;
196  imageCreateInfo.pNext = NULL;
197  imageCreateInfo.imageType = (vk::ImageType)m_Meta->image_type();
198  imageCreateInfo.format = to_vk(m_Meta->format());
199  imageCreateInfo.mipLevels = m_MipLevels;
200  imageCreateInfo.arrayLayers = 1;
201  imageCreateInfo.samples = vk::SampleCountFlagBits::e1;
202  imageCreateInfo.tiling = vk::ImageTiling::eOptimal;
203  imageCreateInfo.usage = vk::ImageUsageFlagBits::eSampled;
204  imageCreateInfo.sharingMode = vk::SharingMode::eExclusive;
205  imageCreateInfo.initialLayout = vk::ImageLayout::eUndefined;
206  imageCreateInfo.extent = vk::Extent3D {m_Meta->width(), m_Meta->height(), m_Meta->depth()};
207  imageCreateInfo.usage = to_vk(m_Meta->usage());
208 
209  utility::vulkan::check(m_Context->device().createImage(&imageCreateInfo, nullptr, &m_Image));
210 
211 
212  // Create and bind the memory
213  vk::MemoryRequirements memReqs;
214  vk::MemoryAllocateInfo memAllocInfo;
215  memAllocInfo.pNext = NULL;
216  memAllocInfo.allocationSize = 0;
217  memAllocInfo.memoryTypeIndex = 0;
218  m_Context->device().getImageMemoryRequirements(m_Image, &memReqs);
219 
220  memAllocInfo.allocationSize = memReqs.size;
221  memAllocInfo.memoryTypeIndex =
222  m_Context->memory_type(memReqs.memoryTypeBits, vk::MemoryPropertyFlagBits::eDeviceLocal);
223 
224  utility::vulkan::check(m_Context->device().allocateMemory(&memAllocInfo, nullptr, &m_DeviceMemory));
225  m_Context->device().bindImageMemory(m_Image, m_DeviceMemory, 0);
226 
227  if(data != nullptr)
228  {
229  vk::CommandBuffer copyCmd = utility::vulkan::create_cmd_buffer(
230  m_Context->device(), m_Context->command_pool(), vk::CommandBufferLevel::ePrimary, true, 1);
231 
232  vk::ImageSubresourceRange subresourceRange;
233  subresourceRange.aspectMask = vk::ImageAspectFlagBits::eColor;
234  subresourceRange.baseMipLevel = 0;
235  subresourceRange.levelCount = m_MipLevels;
236  subresourceRange.layerCount = m_Meta->layers();
237  // Image barrier for optimal image (target)
238  // Optimal image will be used as destination for the copy
239  utility::vulkan::set_image_layout(
240  copyCmd, m_Image, vk::ImageLayout::eUndefined, vk::ImageLayout::eTransferDstOptimal, subresourceRange);
241 
242  // Copy mip levels from staging buffer
243  copyCmd.copyBufferToImage(
244  stagingBuffer->gpu_buffer(), m_Image, vk::ImageLayout::eTransferDstOptimal, (uint32_t)1, &bufferCopyRegion);
245 
246  // Change texture image layout to shader read after all mip levels have been copied
247  m_ImageLayout = vk::ImageLayout::eShaderReadOnlyOptimal;
248 
249  utility::vulkan::set_image_layout(
250  copyCmd, m_Image, vk::ImageLayout::eTransferDstOptimal, m_ImageLayout, subresourceRange);
251 
252  m_Context->flush(copyCmd, true);
253  }
254 
255  // Create image view
256  // Textures are not directly accessed by the shaders and
257  // are abstracted by image views containing additional
258  // information and sub resource ranges
259  vk::ImageViewCreateInfo view;
260  view.image = m_Image;
261  view.viewType = to_vk(m_Meta->image_type());
262  view.format = to_vk(m_Meta->format());
263  view.components = vk::ComponentMapping();
264  view.subresourceRange.aspectMask =
265  (utility::vulkan::has_depth(view.format)) ? vk::ImageAspectFlagBits::eDepth : to_vk(m_Meta->aspect_mask());
266  view.subresourceRange.baseMipLevel = 0;
267  view.subresourceRange.baseArrayLayer = 0;
268  view.subresourceRange.layerCount = m_Meta->layers();
269  // Linear tiling usually won't support mip maps
270  // Only set mip map count if optimal tiling is used
271  view.subresourceRange.levelCount = m_MipLevels;
272  m_SubresourceRange = view.subresourceRange;
273 
274  utility::vulkan::check(m_Context->device().createImageView(&view, nullptr, &m_View));
275 }

278 void texture_t::load_2D()
279 {
280  gli::texture2d* m_Texture2DData = (gli::texture2d*)m_TextureData;
281  if(m_Texture2DData->empty())
282  {
283  LOG_ERROR("Empty texture");
284  debug_break();
285  }
286 
287  if(m_Meta->width() != (uint32_t)(*m_Texture2DData)[0].extent().x)
288  m_Meta->width((uint32_t)(*m_Texture2DData)[0].extent().x);
289 
290  if(m_Meta->height() != (uint32_t)(*m_Texture2DData)[0].extent().y)
291  m_Meta->height((uint32_t)(*m_Texture2DData)[0].extent().y);
292 
293  if(m_Meta->depth() != (uint32_t)(*m_Texture2DData)[0].extent().z)
294  m_Meta->depth((uint32_t)(*m_Texture2DData)[0].extent().z);
295 
296  m_MipLevels = (uint32_t)m_Texture2DData->levels();
297  m_Meta->mip_levels(m_MipLevels);
298  vk::FormatProperties formatProperties;
299 
300  // todo we can map gli::format to the vulkan mappings.
301  if(m_Meta->format() == gfx::format_t::undefined)
302  {
303  LOG_ERROR("Undefined format property in: ", m_Meta->ID().to_string());
304  }
305  m_Context->physical_device().getFormatProperties(to_vk(m_Meta->format()), &formatProperties);
306 
307  // Only use linear tiling if requested (and supported by the device)
308  // Support for linear tiling is mostly limited, so prefer to use
309  // optimal tiling instead
310  // On most implementations linear tiling will only support a very
311  // limited amount of formats and features (mip maps, cubemaps, arrays, etc.)
312  vk::Bool32 useStaging = m_StagingBuffer;
313 
314  // Only use linear tiling if forced
315  if(!useStaging)
316  {
317  // Don't use linear if format is not supported for (linear) shader sampling
318  useStaging = !(formatProperties.linearTilingFeatures & vk::FormatFeatureFlagBits::eSampledImage);
319  }
320 
321  // todo write the version without staging
322  useStaging = true;
323 
324  if(useStaging)
325  {
326  auto stagingBuffer = m_StagingBuffer;
327  if(!m_StagingBuffer)
328  {
329  core::ivk::log->warn(
330  "missing a staging buffer in ivk::texture_t, will create one dynamically, but this is inefficient");
331  auto tempBuffer = m_Cache.create<core::data::buffer_t>(
332  core::gfx::memory_usage::transfer_source,
333  core::gfx::memory_property::host_visible | core::gfx::memory_property::host_coherent,
334  memory::region {m_Texture2DData->size() + 1024, 4, new memory::default_allocator(false)});
335  stagingBuffer = m_Cache.create<ivk::buffer_t>(m_Context, tempBuffer);
336  }
337  if(!stagingBuffer)
338  {
339  core::ivk::log->error("could not create a staging buffer in ivk::texture_t");
340  return;
341  }
342  memory::segment segment;
343  if(auto segmentOpt = stagingBuffer->reserve((vk::DeviceSize)m_Texture2DData->size()); segmentOpt)
344  {
345  segment = segmentOpt.value();
346  gfx::commit_instruction instr;
347  instr.segment = segment;
348  instr.size = (vk::DeviceSize)m_Texture2DData->size();
349  instr.source = (std::uintptr_t)m_Texture2DData->data();
350  if(!stagingBuffer->commit({instr}))
351  {
352  core::ivk::log->error("could not commit an ivk::texture_t in a staging buffer");
353  }
354  }
355  else
356  {
357  core::ivk::log->error("could not allocate a segment in the staging buffer for an ivk::texture_t");
358  return;
359  }
360  // stagingBuffer->map(m_Texture2DData->data(), (vk::DeviceSize)m_Texture2DData->size(), 0);
361 
362  // Setup buffer copy regions for each mip level
363  std::vector<vk::BufferImageCopy> bufferCopyRegions;
364  uintptr_t offset = segment.range().begin;
365 
366  for(uint32_t i = 0; i < m_MipLevels; i++)
367  {
368  vk::BufferImageCopy bufferCopyRegion;
369  bufferCopyRegion.imageSubresource.aspectMask = vk::ImageAspectFlagBits::eColor;
370  bufferCopyRegion.imageSubresource.mipLevel = i;
371  bufferCopyRegion.imageSubresource.baseArrayLayer = 0;
372  bufferCopyRegion.imageSubresource.layerCount = m_Meta->layers();
373  bufferCopyRegion.imageExtent.width = (uint32_t)(*m_Texture2DData)[i].extent().x;
374  bufferCopyRegion.imageExtent.height = (uint32_t)(*m_Texture2DData)[i].extent().y;
375  bufferCopyRegion.imageExtent.depth = 1;
376  bufferCopyRegion.bufferOffset = offset;
377 
378  bufferCopyRegions.push_back(bufferCopyRegion);
379 
380  offset += (*m_Texture2DData)[i].size();
381  }
382 
383  // Create optimal tiled target image
384  vk::ImageCreateInfo imageCreateInfo;
385  imageCreateInfo.pNext = NULL;
386  imageCreateInfo.imageType = (vk::ImageType)m_Meta->image_type();
387  imageCreateInfo.format = to_vk(m_Meta->format());
388  imageCreateInfo.mipLevels = m_MipLevels;
389  imageCreateInfo.arrayLayers = 1;
390  imageCreateInfo.samples = vk::SampleCountFlagBits::e1;
391  imageCreateInfo.tiling = vk::ImageTiling::eOptimal;
392  imageCreateInfo.usage = vk::ImageUsageFlagBits::eSampled;
393  imageCreateInfo.sharingMode = vk::SharingMode::eExclusive;
394  imageCreateInfo.initialLayout = vk::ImageLayout::eUndefined;
395  imageCreateInfo.extent = vk::Extent3D {m_Meta->width(), m_Meta->height(), m_Meta->depth()};
396  imageCreateInfo.usage = to_vk(m_Meta->usage());
397 
398  utility::vulkan::check(m_Context->device().createImage(&imageCreateInfo, nullptr, &m_Image));
399

401  vk::MemoryAllocateInfo memAllocInfo;
402  memAllocInfo.pNext = NULL;
403  memAllocInfo.allocationSize = 0;
404  memAllocInfo.memoryTypeIndex = 0;
405  m_Context->device().getImageMemoryRequirements(m_Image, &memReqs);
406 
407  memAllocInfo.allocationSize = memReqs.size;
408  memAllocInfo.memoryTypeIndex =

132  vk::DeviceSize totalSize = std::accumulate(
133  std::next(std::begin(sizes)), std::end(sizes), sizes[0], [](vk::DeviceSize sum, const vk::DeviceSize& element) {
134  return sum + element;
135  });
136  std::vector<std::pair<memory::segment, memory::range_t>> result;
137 
138  // todo: low priority
139  // this should check for biggest continuous space in the memory::region and split like that
140  if(optimize)

142  auto segment = m_BufferDataHandle->allocate(totalSize);
143  if(segment)
144  {
145  result.resize(sizes.size());
146  vk::DeviceSize accOffset = 0u;
147  for(auto i = 0u; i < sizes.size(); ++i)
148  {
149  result[i].first = segment.value();
150  result[i].second = memory::range_t {accOffset, accOffset + sizes[i]};
151  accOffset += sizes[i];
152  }
153  return result;
154  }
155  }
156 
157  // either optimization failed, or we aren't optimizing
158  result.reserve(sizes.size());
159  for(const auto& size : sizes)
160  {
161  if(auto segment = m_BufferDataHandle->allocate(size); segment)
162  {
163  auto& res = result.emplace_back();
164  res.first = segment.value();
165  res.second = memory::range_t {0, size};
166  }
167  else
168  {
169  goto failure;
170  }
171  }
172 
173  return result;
174 
175  // in case something goes bad, roll back the already completed allocations
176 failure:
177  for(auto& segment : result)
178  {
179  m_BufferDataHandle->deallocate(segment.first);
180  }

184 bool buffer_t::commit(std::vector<core::gfx::commit_instruction> instructions)
185 {
186  PROFILE_SCOPE(core::profiler)
187  vk::DeviceSize totalSize =
188  std::accumulate(std::next(std::begin(instructions)),
189  std::end(instructions),
190  std::begin(instructions)->size,
191  [](vk::DeviceSize sum, const commit_instruction& element) { return sum + element.size; });
192 
193  if(m_BufferDataHandle->memoryPropertyFlags() & core::gfx::memory_property::device_local)
194  {
195  std::vector<vk::DeviceSize> sizeRequests;
196  sizeRequests.reserve(instructions.size());
197  for(const auto& instruction : instructions) sizeRequests.emplace_back(instruction.size);
198 
199  auto stagingBuffer = m_StagingBuffer;
200  if(!stagingBuffer)
201  {
202  core::ivk::log->warn("inefficient loading, dynamically creating a staging ivk::buffer_t.");
203  memory::region temp_region {totalSize, 4, new memory::default_allocator(false)};
204  auto buffer_data = m_Cache.create<core::data::buffer_t>(core::gfx::memory_usage::transfer_source,
205  core::gfx::memory_property::host_visible |
206  core::gfx::memory_property::host_coherent,
207  std::move(temp_region));
208 
209  stagingBuffer = m_Cache.create<core::ivk::buffer_t>(m_Context, buffer_data);
210  }
211  if(!stagingBuffer)
212  {
213  core::ivk::log->error("could not create an ivk::buffer_t for staging.");
214  return false;
215  }
216 
217  auto stagingSegments = stagingBuffer->reserve(sizeRequests, true);
218  if(stagingSegments.size() == 0)
219  {
220  core::ivk::log->error("could not reserve the requested size in the staging buffer.");
221  return false;
222  }
223 
224  if(stagingSegments.size() > 0 && stagingSegments[0].first.range().size() == 0) debug_break();
225 
226  // put the data onto the staging buffer && create the copy region instructions
227  std::vector<vk::BufferCopy> copyRegions;
228  copyRegions.reserve(stagingSegments.size());
229 
230  auto boundSegment = stagingSegments[0].first;
231  std::uintptr_t offset =
232  boundSegment.range().begin - (std::uintptr_t)stagingBuffer->m_BufferDataHandle->region().data();
233  auto tuple = m_Context->device().mapMemory(stagingBuffer->m_Memory, offset, boundSegment.range().size());
234 
235  if(stagingSegments.size() > 0 && stagingSegments[0].first.range().size() == 0) debug_break();
236  for(auto i = 0; i < stagingSegments.size(); ++i)
237  {
238  if(stagingSegments[i].first.range() != boundSegment.range())
239  {
240  m_Context->device().unmapMemory(stagingBuffer->m_Memory);
241  boundSegment = stagingSegments[i].first;
242  offset =
243  boundSegment.range().begin - (std::uintptr_t)stagingBuffer->m_BufferDataHandle->region().data();
244  tuple = m_Context->device().mapMemory(stagingBuffer->m_Memory, offset, boundSegment.range().size());
245  }
246 
247  if(!utility::vulkan::check(tuple.result))
248  {
249  return false;
250  }
251 
252  if(i == 4)
253  {
254  core::log->info("first element: {}", *(float*)(instructions[i].source));
255  }
256 
257  memcpy((void*)((std::uintptr_t)tuple.value + stagingSegments[i].second.begin),
258  (void*)(instructions[i].source),
259  instructions[i].size);
260 
261  if(m_BufferDataHandle->region().allocator()->is_physically_backed())
262  {
263  memcpy((void*)(instructions[i].segment.range().begin +
264  instructions[i].sub_range.value_or(memory::range_t {}).begin),
265  (void*)(instructions[i].source),
266  instructions[i].size);
267  }
268 
269  // todo we can collapse these regions and so invoke less buffercopy commands
270  vk::BufferCopy& copyRegion = copyRegions.emplace_back();
271  copyRegion.srcOffset = offset + stagingSegments[i].second.begin;
272  copyRegion.dstOffset = instructions[i].segment.range().begin +
273  instructions[i].sub_range.value_or(memory::range_t {}).begin -
274  (std::uintptr_t)m_BufferDataHandle->region().data();
275  copyRegion.size = instructions[i].size;
276 
277  if(stagingSegments.size() > 0 && stagingSegments[0].first.range().size() == 0) debug_break();
278  }
279 
280  if(stagingSegments.size() > 0 && stagingSegments[0].first.range().size() == 0) debug_break();
281  m_Context->device().unmapMemory(stagingBuffer->m_Memory);
282  auto res = copy_from(stagingBuffer.value(), copyRegions);
283  for(auto segm : stagingSegments)
284  {
285  if(segm.second.begin == 0) stagingBuffer->deallocate(segm.first);
286  }
287  return res;
288  }
289  else
290  {
291  // core::ivk::log->info("mapping {0} regions into an ivk::buffer_t from CPU.", instructions.size());
292  for(auto& instruction : instructions)
293  {
294  std::uintptr_t offset = instruction.segment.range().begin -
295  (std::uintptr_t)m_BufferDataHandle->region().data() +
296  instruction.sub_range.value_or(memory::range_t {}).begin;
297 
298  auto tuple = m_Context->device().mapMemory(m_Memory, offset, instruction.size);
299  if(!utility::vulkan::check(tuple.result))
300  {
301  return false;
302  }
303  memcpy(tuple.value, (void*)instruction.source, instruction.size);
304  if(m_BufferDataHandle->region().allocator()->is_physically_backed())
305  {
306  memcpy((void*)instruction.segment.range().begin, (void*)instruction.source, instruction.size);
307  }
308  m_Context->device().unmapMemory(m_Memory);
309  }
310  }
311  return true;

316 bool buffer_t::map(const void* data, vk::DeviceSize size, vk::DeviceSize offset)
317 {
318  PROFILE_SCOPE(core::profiler)
319  if(size == 0) return true;
320 
321  if(size > m_BufferDataHandle->size() || (size + offset) > m_BufferDataHandle->size())
322  {
323  core::ivk::log->error("tried to map an incorrect size amount to an ivk::buffer_t.");
324  return false;
325  }
326 
327  if(m_BufferDataHandle->memoryPropertyFlags() & core::gfx::memory_property::device_local)
328  {
329  if(m_StagingBuffer)
330  {
331  m_StagingBuffer->map(data, size, 0);
332  return copy_from(m_StagingBuffer.value(), {vk::BufferCopy {0u, offset, size}});
333  }
334  else
335  {
336  // make a local staging buffer, this is hardly efficient. todo find better way.
337  core::ivk::log->warn("inefficient loading, dynamically creating a staging ivk::buffer_t.");
338  memory::region temp_region {size * 2, 4, new memory::default_allocator(true)};
339  auto buffer_data = m_Cache.create<core::data::buffer_t>(core::gfx::memory_usage::transfer_source,
340  core::gfx::memory_property::host_visible |
341  core::gfx::memory_property::host_coherent,
342  std::move(temp_region));
343  auto staging = m_Cache.create<core::ivk::buffer_t>(m_Context, buffer_data);
344 
345  auto tuple = m_Context->device().mapMemory(staging->m_Memory, 0, size);
346  if(!utility::vulkan::check(tuple.result))
347  {
348  debug_break();
349  }
350  memcpy(tuple.value, data, size);
351  m_Context->device().unmapMemory(staging->m_Memory);
352 
353  return copy_from(staging.value(), {vk::BufferCopy {0u, offset, size}});
354  }
355  }
356  else
357  {
358  // core::ivk::log->info("mapping ivk::buffer_t data from CPU.");
359 
360  auto tuple = m_Context->device().mapMemory(m_Memory, offset, size);
361  if(!utility::vulkan::check(tuple.result))
362  {
363  debug_break();
364  }
365  memcpy(tuple.value, data, size);
366  if(m_BufferDataHandle->region().data() != data) // Let's map this to the buffer too.
367  memcpy(m_BufferDataHandle->region().data(), data, size);
368 
369  m_Context->device().unmapMemory(m_Memory);
370  return true;
371  }

383 {
384  PROFILE_SCOPE(core::profiler)
385  core::profiler.scope_begin("prepare", this);
386  vk::Queue queue = m_Context->queue();
387  wait_until_ready();
388 
389  auto totalsize =
390  std::accumulate(copyRegions.begin(), copyRegions.end(), 0, [&](int sum, const vk::BufferCopy& region) {
391  return sum + (int)region.size;
392  });
393 
394  core::ivk::log->info("copying buffer {0} into {1} for a total size of {2} using {3} copy instructions",
395  utility::to_string(other.m_UID),

397  totalsize,
398  copyRegions.size());
399 
400  for(const auto& region : copyRegions)
401  {
402  core::ivk::log->info(
403  "srcOffset | dstOffset | size : {0} | {1} | {2}", region.srcOffset, region.dstOffset, region.size);
404  }
405 
406  vk::CommandBufferBeginInfo cmdBufferBeginInfo;
407  cmdBufferBeginInfo.pNext = NULL;
408

496  });
497  if(end == std::begin(commands) || end != std::end(commands))

504  for(auto it = end; it != std::end(commands); ++it)
505  message += "size: " + utility::to_string(it->size) + " srcOffset: " + utility::to_string(it->srcOffset) +
506  " dstOffset: " + utility::to_string(it->dstOffset) + "\n";
507  core::ivk::log->debug(message);

516

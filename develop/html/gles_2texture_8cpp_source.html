90  if(m_Texture2DData->empty())

96  GLint internalFormat, format, type;
97

101  static_cast<std::underlying_type_t<gfx::format_t>>(m_Meta->format()));
102  }
103 
104  if(m_Meta->width() != (uint32_t)(*m_Texture2DData)[0].extent().x)
105  m_Meta->width((uint32_t)(*m_Texture2DData)[0].extent().x);
106

109 
110  if(m_Meta->depth() != (uint32_t)(*m_Texture2DData)[0].extent().z)
111  m_Meta->depth((uint32_t)(*m_Texture2DData)[0].extent().z);
112 
113  m_MipLevels = (uint32_t)m_Texture2DData->levels();
114  m_Meta->mip_levels(m_MipLevels);
115 
116  core::igles::log->info("loading texture {} with format {} width {} : height {}",
117  m_Meta->ID().to_string(),
118  m_Meta->format(),
119  m_Meta->width(),
120  m_Meta->height());

162 void texture_t::create_2D(void* data)
163 {
164  glGenTextures(1, &m_Texture);
165  glBindTexture(GL_TEXTURE_2D, m_Texture);
166 
167  GLint internalFormat, format, type;
168  gfx::conversion::to_gles(m_Meta->format(), internalFormat, format, type);
169  auto pixel_storage = gfx::packing_size(m_Meta->format());
170  if(data != nullptr && pixel_storage != 4)
171  {
172  glPixelStorei(GL_UNPACK_ALIGNMENT, pixel_storage);
173  }
174 
175  glTexStorage2D(GL_TEXTURE_2D, 1, internalFormat, m_Meta->width(), m_Meta->height());
176 
177  if(data != nullptr)
178  {
179  if(type != 0)
180  {
181  glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, m_Meta->width(), m_Meta->height(), format, type, data);
182  }
183  else
184  {
185  assert_debug_break(false);
186  }
187 
188  if(pixel_storage != 4) glPixelStorei(GL_UNPACK_ALIGNMENT, 4);

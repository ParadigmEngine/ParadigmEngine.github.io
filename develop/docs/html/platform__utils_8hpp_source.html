140  static psl::string to_unix(psl::string_view path)
141  {
142  psl::string dir;
143  while(path.size() > 0 && (path[0] == '\"' || path[0] == '\''))
144  {
145  path = path.substr(1);
146  }
147 
148  while(path.size() > 0 && (path[path.size() - 1] == '\"' || path[path.size() - 1] == '\''))
149  {
150  path = path.substr(0, path.size() - 1);
151  }
152 
153  if(auto find = path.find(":\\\\"); find == 1u)
154  {
155  psl::char_t drive = utility::string::to_lower(path.substr(0u, 1u))[0];
156  dir = "/mnt/" + psl::string(1, drive) + seperator + path.substr(4u);
157  }
158  else if(find = path.find(":\\"); find == 1u)
159  {
160  psl::char_t drive = utility::string::to_lower(path.substr(0u, 1u))[0];
161  dir = "/mnt/" + psl::string(1, drive) + seperator + path.substr(3u);
162  }
163  else if(find = path.find(":/"); find == 1u)
164  {
165  psl::char_t drive = utility::string::to_lower(path.substr(0u, 1u))[0];
166  dir = "/mnt/" + psl::string(1, drive) + seperator + path.substr(3u);
167  }
168  else
169  {
170  dir = path;
171  }
172 
173  auto find = dir.find('\\');
174  while(find < dir.size())
175  {
176  dir.replace(find, 1, "/");
177  find = dir.find('\\');
178  }
179 
180  return dir;
181  }

186  static psl::string to_windows(psl::string_view path)
187  {
188  psl::string dir;
189  while(path.size() > 0 && (path[0] == '\"' || path[0] == '\''))
190  {
191  path = path.substr(1);
192  }
193 
194  while(path.size() > 0 && (path[path.size() - 1] == '\"' || path[path.size() - 1] == '\''))
195  {
196  path = path.substr(0, path.size() - 1);
197  }
198 
199  if(path[0] == '/')
200  {
201  dir = psl::string(path.substr(5u, 1u)) + ":/" + path.substr(7u);
202  }
203  else
204  {
205  dir = path;
206  }
207 
208  auto find = dir.find('\\');
209  while(find < dir.size())
210  {
211  dir.replace(find, 1, "/");
212  find = dir.find('\\');
213  }
214 
215  find = dir.find("//");
216  while(find < dir.size())
217  {
218  dir.erase(find, 1);
219  find = dir.find("//");
220  }
221  return dir;
222  }

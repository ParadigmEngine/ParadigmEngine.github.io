57 void geometry_t::recreate(core::resource::handle<core::data::geometry_t> data)
58 {
59  clear();
60  std::vector<vk::DeviceSize> sizeRequests;
61  sizeRequests.reserve(data->vertex_streams().size() + ((m_GeometryBuffer == m_IndicesBuffer) ? 1 : 0));
62  std::for_each(std::begin(data->vertex_streams()),
63  std::end(data->vertex_streams()),
64  [&sizeRequests](const std::pair<psl::string, core::stream>& element) {
65  sizeRequests.emplace_back((uint32_t)element.second.bytesize());
66  });
67 
68  if(m_GeometryBuffer == m_IndicesBuffer)
69  {
70  sizeRequests.emplace_back((uint32_t)(data->indices().size() * sizeof(core::data::geometry_t::index_size_t)));
71  }
72 
73  auto segments = m_GeometryBuffer->reserve(sizeRequests, true);
74  if(segments.size() == 0)
75  {
76  core::ivk::log->critical("ran out of memory, could not allocate enough in the buffer to accomodate");
77  exit(1);
78  }
79  assert(data->vertex_streams().size() > 0);
80  m_Vertices = std::begin(data->vertex_streams())->second.size();
81  std::vector<core::gfx::commit_instruction> instructions;
82  size_t i = 0;
83  for(const auto& stream : data->vertex_streams())
84  {
85  assert(m_Vertices == stream.second.size());
86  auto& instr = instructions.emplace_back();
87  instr.size = stream.second.bytesize();
88  instr.source = (std::uintptr_t)(stream.second.cdata());
89  instr.segment = segments[i].first;
90  if(segments[i].first.range() != segments[i].second) instr.sub_range = segments[i].second;
91 
92  auto& b = m_Bindings.emplace_back();
93  b.name = stream.first;
94  b.segment = segments[i].first;
95  b.sub_range = segments[i].second;
96  ++i;
97  }
98 
99  if(m_GeometryBuffer != m_IndicesBuffer)
100  {
101  if(auto indiceSegment =
102  m_IndicesBuffer->reserve((uint32_t)(data->indices().size() * sizeof(core::data::geometry_t::index_size_t)));
103  indiceSegment)
104  {
105  m_IndicesSegment = indiceSegment.value();
106  m_IndicesSubRange = memory::range_t {0, m_IndicesSegment.range().size()};
107  }
108  else
109  {
110  core::ivk::log->critical("index buffer was out of memory");
111  // todo error condition could not allocate segment
112  exit(1);
113  }
114  gfx::commit_instruction instr;
115  instr.size = m_IndicesSegment.range().size();
116  instr.source = (std::uintptr_t)data->indices().data();
117  instr.segment = m_IndicesSegment;
118  m_IndicesBuffer->commit({instr});
119  }
120  else
121  {
122  i = segments.size() - 1;
123  auto& instr = instructions.emplace_back();
124  instr.size = segments[i].second.size();
125  instr.source = (std::uintptr_t)data->indices().data();
126  instr.segment = segments[i].first;
127  if(segments[i].first.range() != segments[i].second) instr.sub_range = segments[i].second;
128 
129  m_IndicesSegment = segments[i].first;
130  m_IndicesSubRange = segments[i].second;
131  }
132 
133  m_Triangles = (m_IndicesSubRange.size() / sizeof(core::data::geometry_t::index_size_t)) / 3u;
134 
135  m_GeometryBuffer->commit(instructions);
136  m_Data = data;
137 }

168  success:

181 {

187 
188  auto binding = std::find_if(
189  std::begin(m_Bindings), std::end(m_Bindings), [tag = attribute.tag()](const auto& binding) noexcept {
190  return binding.name == tag.data();
191  });
192 
193  auto offset = vk::DeviceSize {binding->segment.range().begin + binding->sub_range.begin};
194  buffer.bindVertexBuffers(attribute.location(), 1, &m_GeometryBuffer->gpu_buffer(), &offset);
195  }
196  }
197 
198  buffer.bindIndexBuffer(
199  m_IndicesBuffer->gpu_buffer(), m_IndicesSegment.range().begin + m_IndicesSubRange.begin, INDEX_TYPE);

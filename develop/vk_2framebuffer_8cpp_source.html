20  m_Context(context),
21  m_Data(data) // , m_Data(copy(cache, data))
22 {
23  m_Framebuffers.resize(m_Data->framebuffers());
24  size_t index = 0u;
25  for(const auto& attach : m_Data->attachments())
26  {
27  if(!add(cache,
28  attach.texture(),
29  gfx::conversion::to_vk(attach),
30  index,
31  (attach.shared()) ? 1u : m_Framebuffers.size()))
32  throw std::runtime_error("");
33  index += m_Framebuffers.size();
34  }
35 
36  m_Sampler = cache.find<core::ivk::sampler_t>(m_Data->sampler().value());
37  if(!m_Sampler)
38  {
39  core::ivk::log->error("could not load sampler {0} for framebuffer {1}", m_Sampler.uid(), metaData.uid);
40  assert(false);
41  }
42 
43  // now we create the renderpass that describes the framebuffer
44  std::vector<vk::AttachmentDescription> attachmentDescriptions;
45  std::transform(std::begin(m_Bindings),
46  std::end(m_Bindings),
47  std::back_inserter(attachmentDescriptions),
48  [](const auto& binding) { return binding.description; });
49 
50  // Collect attachment references
51  std::vector<vk::AttachmentReference> colorReferences;
52  vk::AttachmentReference depthReference;
53  bool hasDepth = false;
54  bool hasColor = false;
55 
56  uint32_t attachmentIndex = 0;
57 
58  for(auto& binding : m_Bindings)
59  {
60  if(utility::vulkan::has_depth(binding.description.format) ||
61  utility::vulkan::has_stencil(binding.description.format))
62  {
63  // Only one depth attachment allowed
64  // assert(!hasDepth);
65  depthReference.attachment = attachmentIndex;
66  depthReference.layout = vk::ImageLayout::eDepthStencilAttachmentOptimal;
67  hasDepth = true;
68  }
69  else
70  {
71  colorReferences.push_back({attachmentIndex, vk::ImageLayout::eColorAttachmentOptimal});
72  hasColor = true;
73  }
74  attachmentIndex++;
75  };
76 
77  // Default render pass setup uses only one subpass
78  vk::SubpassDescription subpass;
79  subpass.pipelineBindPoint = vk::PipelineBindPoint::eGraphics;
80  if(hasColor)
81  {
82  subpass.pColorAttachments = colorReferences.data();
83  subpass.colorAttachmentCount = static_cast<uint32_t>(colorReferences.size());
84  }
85  if(hasDepth)
86  {
87  subpass.pDepthStencilAttachment = &depthReference;
88  }
89 
90  // Use subpass dependencies for attachment layout transitions
91  std::array<vk::SubpassDependency, 2> dependencies;
92 
93  dependencies[0].srcSubpass = VK_SUBPASS_EXTERNAL;
94  dependencies[0].dstSubpass = 0;
95  dependencies[0].srcStageMask = vk::PipelineStageFlagBits::eBottomOfPipe;
96  dependencies[0].dstStageMask = vk::PipelineStageFlagBits::eColorAttachmentOutput;
97  dependencies[0].srcAccessMask = vk::AccessFlagBits::eMemoryRead;
98  dependencies[0].dstAccessMask =
99  vk::AccessFlagBits::eColorAttachmentRead | vk::AccessFlagBits::eColorAttachmentWrite;
100  dependencies[0].dependencyFlags = vk::DependencyFlagBits::eByRegion;
101 
102  dependencies[1].srcSubpass = 0;
103  dependencies[1].dstSubpass = VK_SUBPASS_EXTERNAL;
104  dependencies[1].srcStageMask = vk::PipelineStageFlagBits::eColorAttachmentOutput;
105  dependencies[1].dstStageMask = vk::PipelineStageFlagBits::eBottomOfPipe;
106  dependencies[1].srcAccessMask =
107  vk::AccessFlagBits::eColorAttachmentRead | vk::AccessFlagBits::eColorAttachmentWrite;
108  dependencies[1].dstAccessMask = vk::AccessFlagBits::eMemoryRead;
109  dependencies[1].dependencyFlags = vk::DependencyFlagBits::eByRegion;
110 
111  // Create render pass
112  vk::RenderPassCreateInfo renderPassInfo;
113  renderPassInfo.pAttachments = attachmentDescriptions.data();
114  renderPassInfo.attachmentCount = static_cast<uint32_t>(attachmentDescriptions.size());
115  renderPassInfo.subpassCount = 1;
116  renderPassInfo.pSubpasses = &subpass;
117  renderPassInfo.dependencyCount = 2;
118  renderPassInfo.pDependencies = dependencies.data();
119  utility::vulkan::check(m_Context->device().createRenderPass(&renderPassInfo, nullptr, &m_RenderPass));
120 
121  // pre-allocate attachmentviews, and fill in all the data in the FBCI
122  // we will override the views in the upcoming loop, where the device also will create the FBOs
123  std::vector<vk::ImageView> attachmentViews;
124  attachmentViews.resize(m_Bindings.size());
125 
126  vk::FramebufferCreateInfo framebufferInfo;
127  framebufferInfo.renderPass = m_RenderPass;
128  framebufferInfo.pAttachments = attachmentViews.data();
129  framebufferInfo.attachmentCount = static_cast<uint32_t>(attachmentViews.size());
130  framebufferInfo.width = m_Data->width();
131  framebufferInfo.height = m_Data->height();
132  framebufferInfo.layers = 0;
133 
134  for(auto i = 0u; i < m_Data->framebuffers(); ++i)
135  {
136  // Find. max number of layers across attachments && store all attachment views
137  uint32_t maxLayers = 0u;
138 
139  for(const auto& binding : m_Bindings)
140  {
141  attachmentViews[binding.index] = binding.attachments[i]->view();
142  maxLayers = std::max(maxLayers, binding.attachments[i]->subResourceRange().layerCount);
143  }
144 
145  framebufferInfo.layers = maxLayers;
146  utility::vulkan::check(m_Context->device().createFramebuffer(&framebufferInfo, nullptr, &m_Framebuffers[i]));
147  }
148 }

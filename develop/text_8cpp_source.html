141 core::resource::handle<core::data::geometry_t> text::create_text(psl::string_view text)
142 {
143  core::stream vertStream {core::stream::type::vec3};
144  core::stream normStream {core::stream::type::vec3};
145  core::stream colorStream {core::stream::type::vec4};
146  core::stream uvStream {core::stream::type::vec2};
147 
148  auto& vertices = vertStream.as_vec3().value().get();
149  auto& normals = normStream.as_vec3().value().get();
150  auto& colors = colorStream.as_vec4().value().get();
151  auto& uvs = uvStream.as_vec2().value().get();
152 
153  float right = 0;
154  float up = 1.0f;
155 
156  // validate for illegal characters in input.
157  {
158  const auto max_char = character_data.size() + 32;
159  for(int character : text)
160  {
161  assert((character >= 32 && character < max_char) || character == '\n' || character == '\t');
162  }
163  }
164 
165  auto size = std::accumulate(std::begin(text), std::end(text), size_t {0u}, [](size_t sum, int character) -> size_t {
166  return sum + ((character == '\n') ? 0u : (character == '\t') ? 4u : 1u);
167  });
168 
169  vertices.reserve(size * 4);
170  uvs.reserve(size * 4);
171 
172  auto insert_character = [&vertices, &uvs](float& right, float up, const auto& data) {
173  vertices.emplace_back(psl::vec3 {right + data.quad[0], up - data.quad[3], 0.0f});
174  vertices.emplace_back(psl::vec3 {right + data.quad[2], up - data.quad[3], 0.0f});
175  vertices.emplace_back(psl::vec3 {right + data.quad[2], up - data.quad[1], 0.0f});
176  vertices.emplace_back(psl::vec3 {right + data.quad[0], up - data.quad[1], 0.0f});
177 
178  uvs.emplace_back(psl::vec2 {data.uv[0], data.uv[3]});
179  uvs.emplace_back(psl::vec2 {data.uv[2], data.uv[3]});
180  uvs.emplace_back(psl::vec2 {data.uv[2], data.uv[1]});
181  uvs.emplace_back(psl::vec2 {data.uv[0], data.uv[1]});
182 
183  right += data.offset.x;
184  };
185 
186  size_t text_count {0u};
187  for(int character : text)
188  {
189  if(character == '\n')
190  {
191  right = 0.0f;
192  up -= 1.0f;
193  text_count = 0;
194  continue;
195  }
196 
197  if(character == '\t')
198  {
199  character = ' ' - 32;
200  const auto& data = character_data[character];
201 
202  for(auto i = 4 - text_count % 4; i > 0; --i)
203  {
204  insert_character(right, up, data);
205  ++text_count;
206  }
207  }
208  else
209  {
210  const auto& data = character_data[character - 32];
211  insert_character(right, up, data);
212  ++text_count;
213  }
214  }

217 
218  normals.resize(vertices.size());

220  colors.resize(vertices.size());

226  geomData->vertices(core::data::geometry_t::constants::NORMAL, normStream);

228  geomData->vertices(core::data::geometry_t::constants::TEX, uvStream);
229

100 bool drawpass::build()
101 {
102  LOG_INFO("Rebuilding Command Buffers");
103  m_LastBuildFrame = m_FrameCount;
104 
105  m_Context->device().waitIdle();
106  m_Context->device().freeCommandBuffers(
107  m_Context->command_pool(), (uint32_t)m_DrawCommandBuffers.size(), m_DrawCommandBuffers.data());
108 
109  m_Buffers = (uint32_t)m_DrawCommandBuffers.size();
110 
111  vk::CommandBufferAllocateInfo cmdBufAllocateInfo;
112  cmdBufAllocateInfo.commandPool = m_Context->command_pool();
113  cmdBufAllocateInfo.commandBufferCount = (uint32_t)m_DrawCommandBuffers.size(); // one for each image
114  cmdBufAllocateInfo.level = vk::CommandBufferLevel::ePrimary;
115 
116  if(!utility::vulkan::check(
117  m_Context->device().allocateCommandBuffers(&cmdBufAllocateInfo, m_DrawCommandBuffers.data())))
118  throw new std::runtime_error("Critical issue");
119 
120  vk::CommandBufferBeginInfo cmdBufInfo;
121  cmdBufInfo.pNext = NULL;
122 
123  std::vector<vk::ClearValue> clearValues;
124 
125  if(m_UsingSwap)
126  {
127  clearValues.reserve(1 + (size_t)m_Swapchain->has_depth());
128  for(auto i = 0; i < 1; ++i)
129  {
130  clearValues.emplace_back(m_Swapchain->clear_color());
131  }
132  if(m_Swapchain->has_depth())
133  {
134  clearValues.emplace_back(m_Swapchain->clear_depth());
135  }
136  }
137  else
138  {
139  const auto& attachments = m_Framebuffer->data()->attachments();
140  std::transform(std::begin(attachments),
141  std::end(attachments),
142  std::back_inserter(clearValues),
143  [](const auto& attach) { return core::gfx::conversion::to_vk(attach.clear_value()); });
144  }
145  bool success = false;
146  for(auto i = 0; i < m_DrawCommandBuffers.size(); ++i)
147  {
148  m_Context->device().waitIdle();
149  if(!utility::vulkan::check(m_DrawCommandBuffers[i].begin(cmdBufInfo)))
150  throw new std::runtime_error("Critical issue");
151 
152  const std::vector<vk::Framebuffer>& framebuffers =
153  (m_UsingSwap) ? m_Swapchain->framebuffers() : m_Framebuffer->framebuffers();
154 
155 
156  vk::RenderPassBeginInfo renderPassBeginInfo;
157 
158  if(m_UsingSwap)
159  {
160  renderPassBeginInfo.renderPass = m_Swapchain->renderpass();
161  renderPassBeginInfo.renderArea.extent.width = m_Swapchain->width();
162  renderPassBeginInfo.renderArea.extent.height = m_Swapchain->height();
163  }
164  else
165  {
166  renderPassBeginInfo.renderPass = m_Framebuffer->render_pass();
167  renderPassBeginInfo.renderArea.extent.width = m_Framebuffer->data()->width();
168  renderPassBeginInfo.renderArea.extent.height = m_Framebuffer->data()->height();
169  }
170 
171  renderPassBeginInfo.renderArea.offset.x = 0;
172  renderPassBeginInfo.renderArea.offset.y = 0;
173  renderPassBeginInfo.clearValueCount = (uint32_t)clearValues.size();
174  renderPassBeginInfo.pClearValues = clearValues.data();
175  if(m_UsingSwap)
176  {
177  renderPassBeginInfo.framebuffer = framebuffers[i];
178 
179  m_DrawCommandBuffers[i].beginRenderPass(renderPassBeginInfo, vk::SubpassContents::eInline);
180 
181  // Update dynamic viewport state
182  vk::Viewport viewport;
183  viewport.height = (float)renderPassBeginInfo.renderArea.extent.height;
184  viewport.width = (float)renderPassBeginInfo.renderArea.extent.width;
185  viewport.minDepth = (float)0.0f;
186  viewport.maxDepth = (float)1.0f;
187  m_DrawCommandBuffers[i].setViewport(0, 1, &viewport);
188 
189  // Update dynamic scissor state
190  vk::Rect2D scissor;
191  scissor.extent.width = renderPassBeginInfo.renderArea.extent.width;
192  scissor.extent.height = renderPassBeginInfo.renderArea.extent.height;
193  scissor.offset.x = 0;
194  scissor.offset.y = 0;
195  m_DrawCommandBuffers[i].setScissor(0, 1, &scissor);
196 
197  m_DrawCommandBuffers[i].setDepthBias(
198  m_DepthBias.components[0], m_DepthBias.components[1], m_DepthBias.components[2]);
199 
200  for(auto& group : m_AllGroups) build_drawgroup(group, m_DrawCommandBuffers[i], m_Swapchain, i);
201 
202  m_DrawCommandBuffers[i].endRenderPass();
203  }
204  else
205  {
206  for(const auto& framebuffer : framebuffers)
207  {
208  renderPassBeginInfo.framebuffer = framebuffer;
209 
210  m_DrawCommandBuffers[i].beginRenderPass(renderPassBeginInfo, vk::SubpassContents::eInline);
211 
212  // Update dynamic viewport state
213  vk::Viewport viewport;
214  viewport.height = (float)renderPassBeginInfo.renderArea.extent.height;
215  viewport.width = (float)renderPassBeginInfo.renderArea.extent.width;
216  viewport.minDepth = (float)0.0f;
217  viewport.maxDepth = (float)1.0f;
218  m_DrawCommandBuffers[i].setViewport(0, 1, &viewport);
219 
220  // Update dynamic scissor state
221  vk::Rect2D scissor;
222  scissor.extent.width = renderPassBeginInfo.renderArea.extent.width;
223  scissor.extent.height = renderPassBeginInfo.renderArea.extent.height;
224  scissor.offset.x = 0;
225  scissor.offset.y = 0;
226  m_DrawCommandBuffers[i].setScissor(0, 1, &scissor);
227 
228  m_DrawCommandBuffers[i].setDepthBias(
229  m_DepthBias.components[0], m_DepthBias.components[1], m_DepthBias.components[2]);
230 
231 
232  for(auto& group : m_AllGroups) build_drawgroup(group, m_DrawCommandBuffers[i], m_Framebuffer, i);
233 
234 
235  m_DrawCommandBuffers[i].endRenderPass();
236  }
237  }
238 
239  success |= utility::vulkan::check(m_DrawCommandBuffers[i].end());
240  }
241 
242  return success;

257 void drawpass::destroy_fences()
258 {
259  if(!utility::vulkan::check(m_Context->device().waitForFences(
260  m_WaitFences.size(), m_WaitFences.data(), VK_TRUE, std::numeric_limits<uint64_t>::max())))
261  LOG_ERROR("Failed to wait for fence");
262 
263  if(!utility::vulkan::check(m_Context->device().resetFences(m_WaitFences.size(), m_WaitFences.data())))
264  LOG_ERROR("Failed to reset fence");

286 void drawpass::present()
287 {
288  if(m_WaitFences.size() > 0)
289  {
290  if(!utility::vulkan::check(m_Context->device().waitForFences(
291  1, &m_WaitFences[m_CurrentBuffer], VK_TRUE, std::numeric_limits<uint64_t>::max())))
292  LOG_ERROR("Failed to wait for fence");
293 
294  if(!utility::vulkan::check(m_Context->device().resetFences(1, &m_WaitFences[m_CurrentBuffer])))
295  LOG_ERROR("Failed to reset fence");
296  }
297 
298  std::vector<vk::Semaphore> semaphores {m_WaitFor};
299  std::vector<vk::PipelineStageFlags> stageFlags;
300 
301  if(m_UsingSwap)
302  {
303  semaphores.push_back(m_PresentComplete);
304  }
305 
306  if(semaphores.size() == 0)
307  {
308  m_SubmitInfo.pWaitSemaphores = VK_NULL_HANDLE;
309  m_SubmitInfo.waitSemaphoreCount = 0;
310  }
311  else
312  {
313  m_SubmitInfo.waitSemaphoreCount = (uint32_t)semaphores.size();
314  m_SubmitInfo.pWaitSemaphores = semaphores.data();
315 
316  for(auto i = 0; i < semaphores.size(); ++i) stageFlags.push_back(m_SubmitPipelineStages);
317 
318  m_SubmitInfo.pWaitDstStageMask = stageFlags.data();
319  }
320 
321  // Signal ready with offscreen semaphore
322  m_SubmitInfo.pSignalSemaphores = &m_RenderComplete;
323 
324  assert(m_DrawCommandBuffers[m_CurrentBuffer]);
325  m_SubmitInfo.pCommandBuffers = &m_DrawCommandBuffers[m_CurrentBuffer];
326  m_SubmitInfo.commandBufferCount = 1;
327 
328 
329  if(m_WaitFences.size() > 0)
330  utility::vulkan::check(m_Context->queue().submit(1, &m_SubmitInfo, m_WaitFences[m_CurrentBuffer]));
331  else
332  utility::vulkan::check(m_Context->queue().submit(1, &m_SubmitInfo, nullptr));
333 
334  if(m_UsingSwap)
335  {
336  utility::vulkan::check(m_Swapchain->present(m_RenderComplete));
337  }
338 
339  ++m_FrameCount;
340 }

368 void drawpass::build_drawgroup(drawgroup& group,
369  vk::CommandBuffer cmdBuffer,
370  core::resource::handle<core::ivk::framebuffer_t> framebuffer,
371  uint32_t index)
372 {
373  for(auto& drawLayer : group.m_Group)
374  {
375  psl::array<uint32_t> render_indices;
376  for(auto& drawCall : drawLayer.second)
377  {
378  auto matIndices = drawCall.m_Bundle->materialIndices(drawLayer.first.begin(), drawLayer.first.end());
379  render_indices.insert(std::end(render_indices), std::begin(matIndices), std::end(matIndices));
380  }
381  std::sort(std::begin(render_indices), std::end(render_indices));
382  render_indices.erase(std::unique(std::begin(render_indices), std::end(render_indices)),
383  std::end(render_indices));
384  for(auto renderLayer : render_indices)
385  {
386  for(auto& drawCall : drawLayer.second)
387  {
388  if(drawCall.m_Geometry.size() == 0 || !drawCall.m_Bundle->has(renderLayer)) continue;
389  auto bundle = drawCall.m_Bundle;
390  if(!bundle->bind_material(renderLayer)) continue;
391  auto gfxmat {bundle->bound()};
392  auto mat {gfxmat->resource<gfx::graphics_backend::vulkan>()};
393 
394  if(!mat->bind_pipeline(cmdBuffer, framebuffer, index)) continue;
395  for(auto& [gfxGeometryHandle, count] : drawCall.m_Geometry)
396  {
397  auto geometryHandle = gfxGeometryHandle->resource<gfx::graphics_backend::vulkan>();
398  uint32_t instance_n = bundle->instances(gfxGeometryHandle);
399  if(instance_n == 0 || !geometryHandle->compatible(mat.value())) continue;
400 
401  geometryHandle->bind(cmdBuffer, mat.value());
402 
403  // bundle->bind_geometry(cmdBuffer, geometryHandle);
404 
405  for(const auto& b : bundle->m_InstanceData.bindings(gfxmat, gfxGeometryHandle))
406  {
407  cmdBuffer.bindVertexBuffers(b.first,
408  1,
409  &bundle->m_InstanceData.vertex_buffer()
410  ->resource<gfx::graphics_backend::vulkan>()
411  ->gpu_buffer(),
412  &b.second);
413  }
414 
415  cmdBuffer.drawIndexed((uint32_t)geometryHandle->data()->indices().size(), instance_n, 0, 0, 0);
416  }
417  }
418  }
419  }
420 }

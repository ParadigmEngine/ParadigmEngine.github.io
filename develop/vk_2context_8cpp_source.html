26 VKAPI_ATTR VkBool32 VKAPI_CALL VulkanDebugCB(VkDebugReportFlagsEXT flags,
27  VkDebugReportObjectTypeEXT objectType,
28  uint64_t object,
29  size_t location,
30  int32_t messageCode,
31  const char* layer_prefix,
32  const char* msg,
33  void* pUserData)
34 {
35  psl::string message {msg};
36  message.append(" [");
37  message.append(layer_prefix);
38  message.append("]");
39  switch(flags)
40  {
41  case VK_DEBUG_REPORT_FLAG_BITS_MAX_ENUM_EXT:
42  core::ivk::log->info(message.c_str());
43  break;
44  case VK_DEBUG_REPORT_PERFORMANCE_WARNING_BIT_EXT:
45  core::ivk::log->warn(message.c_str());
46  break;
47  case VK_DEBUG_REPORT_WARNING_BIT_EXT:
48  core::ivk::log->warn(message.c_str());
49  core::ivk::log->flush();
50  break;
51  case VK_DEBUG_REPORT_ERROR_BIT_EXT:
52  core::ivk::log->error(message.c_str());
53  core::ivk::log->flush();
54  break;
55  case VK_DEBUG_REPORT_DEBUG_BIT_EXT:
56  core::ivk::log->debug(message.c_str());
57  break;
58  case VK_DEBUG_REPORT_INFORMATION_BIT_EXT:
59  default:
60  core::ivk::log->info(message.c_str());
61  break;
62  }
63  return false; // always return false
64 }

66 inline psl::string8_t size_denotation(size_t size)
67 {
68  static const std::vector<psl::string8::view> SUFFIXES {{"B", "KB", "MB", "GB", "TB", "PB"}};
69  size_t suffixIndex = 0;
70  while(suffixIndex < SUFFIXES.size() - 1 && size > 1024)
71  {
72  size >>= 10;
73  ++suffixIndex;
74  }
75 
76  psl::string8::stream buffer;
77  buffer << size << " " << SUFFIXES[suffixIndex];
78  return buffer.str();
79 }

81 struct VKAPIVersion
82 {
83  public:
84  explicit VKAPIVersion(uint32_t version) :
85  major((uint32_t)(version) >> 22), minor(((uint32_t)(version) >> 12) & 0x3ff),
86  patch((uint32_t)(version)&0xfff) {};
87 
88  bool operator==(const VKAPIVersion& other) const
89  {
90  return (major == other.major && minor == other.minor && patch == other.patch);
91  };
92  bool operator!=(const VKAPIVersion& other) const
93  {
94  return (major != other.major || minor != other.minor || patch != other.patch);
95  };
96  bool operator>(const VKAPIVersion& other) const
97  {
98  return (major > other.major && minor > other.minor && patch > other.patch);
99  };
100  bool operator>=(const VKAPIVersion& other) const
101  {
102  return (major >= other.major && minor >= other.minor && patch >= other.patch);
103  };
104  bool operator<(const VKAPIVersion& other) const
105  {
106  return (major < other.major && minor < other.minor && patch < other.patch);
107  };
108  bool operator<=(const VKAPIVersion& other) const
109  {
110  return (major <= other.major && minor <= other.minor && patch <= other.patch);
111  };
112  const uint32_t major;
113  const uint32_t minor;
114  const uint32_t patch;
115 };

607 bool context::queue_index(vk::QueueFlags flag, vk::Queue& queue, uint32_t& queueIndex)
608 {
609  // Find a queue that supports graphics operations
610  std::vector<vk::QueueFamilyProperties> queueProps = m_PhysicalDevice.getQueueFamilyProperties();
611  std::optional<uint32_t> secondBest = std::nullopt;
612  for(queueIndex = 0; queueIndex < queueProps.size(); queueIndex++)
613  {
614  if(queueProps[queueIndex].queueFlags & flag) secondBest = queueIndex;
615 
616  if(queueProps[queueIndex].queueFlags == flag) break;
617  }
618  if(queueIndex >= queueProps.size() && secondBest.has_value()) queueIndex = secondBest.value();
619  if(queueIndex >= queueProps.size())
620  {
621  core::ivk::log->warn("Could not find the appropriate queue enabled bit for {}", vk::to_string(flag));
622  return false;
623  }
624  return true;

675

679  }

688  if(!queue_index(vk::QueueFlagBits::eGraphics, m_Queue, m_GraphicsQueueIndex))

700  queueCreateInfo[0].queueFamilyIndex = m_GraphicsQueueIndex;
701  queueCreateInfo[0].queueCount = 1;
702  queueCreateInfo[0].pQueuePriorities = queuePriorities.data();
703 
704  if(!utility::vulkan::check(create_device(queueCreateInfo.data(), (uint32_t)queueCreateInfo.size(), m_Device)))

708  }
709 
710  m_Device.getQueue(m_GraphicsQueueIndex, 0, &m_Queue);
711  m_TransferQueue = m_Queue;
712  }
713  else
714  {
715  // Vulkan device
716  std::array<float, 1> queuePriorities = {1.0f};
717  queueCreateInfo.resize(2);
718  queueCreateInfo[0].queueFamilyIndex = m_GraphicsQueueIndex;
719  queueCreateInfo[0].queueCount = 1;
720  queueCreateInfo[0].pQueuePriorities = queuePriorities.data();
721 
722  queueCreateInfo[1].queueFamilyIndex = m_TransferQueueIndex;
723  queueCreateInfo[1].queueCount = 1;
724  queueCreateInfo[1].pQueuePriorities = queuePriorities.data();
725 
726 
727  if(!utility::vulkan::check(create_device(queueCreateInfo.data(), (uint32_t)queueCreateInfo.size(), m_Device)))
728  {
729  core::ivk::log->critical("Could not create a Vulkan device.");
730  std::exit(-1);
731  }
732 
733  m_Device.getQueue(m_GraphicsQueueIndex, 0, &m_Queue);
734  m_Device.getQueue(m_TransferQueueIndex, 0, &m_TransferQueue);
735  }

737 #if !defined(VK_STATIC)

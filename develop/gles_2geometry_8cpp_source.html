61 {
62  clear(false);
63 
64  psl::array<size_t> sizeRequests;
65  sizeRequests.reserve(data->vertex_streams().size() + ((m_GeometryBuffer == m_IndicesBuffer) ? 1 : 0));
66  std::for_each(std::begin(data->vertex_streams()),
67  std::end(data->vertex_streams()),
68  [&sizeRequests](const std::pair<psl::string, core::stream>& element) {
69  sizeRequests.emplace_back(element.second.bytesize());
70  });
71  auto error = glGetError();
72 
73  if(m_GeometryBuffer == m_IndicesBuffer)
74  {
75  sizeRequests.emplace_back((uint32_t)(data->indices().size() * sizeof(core::data::geometry_t::index_size_t)));
76  }
77 
78  auto segments = m_GeometryBuffer->allocate(sizeRequests, true);
79  if(segments.size() == 0)
80  {
81  core::igles::log->critical("ran out of memory, could not allocate enough in the buffer to accomodate");
82  exit(1);
83  }
84  assert(data->vertex_streams().size() > 0);
85  m_Vertices = std::begin(data->vertex_streams())->second.size();
86  std::vector<core::gfx::memory_copy> instructions;
87  auto current_segment = std::begin(segments);
88  for(const auto& stream : data->vertex_streams())
89  {
90  assert(m_Vertices == stream.second.size());
91  auto& instr = instructions.emplace_back();
92  instr.size = stream.second.bytesize();
93  instr.destination_offset = current_segment->first.range().begin + current_segment->second.begin;
94  instr.source_offset = (std::uintptr_t)(stream.second.cdata());
95 
96  auto& b = m_Bindings.emplace_back();
97  b.name = stream.first;
98  b.segment = current_segment->first;
99  b.sub_range = current_segment->second;
100  current_segment = std::next(current_segment);
101  }
102 
103  if(m_GeometryBuffer != m_IndicesBuffer)
104  {
105  if(auto indiceSegment =
106  m_IndicesBuffer->allocate(data->indices().size() * sizeof(core::data::geometry_t::index_size_t));
107  indiceSegment)
108  {
109  m_IndicesSegment = indiceSegment.value();
110  m_IndicesSubRange = memory::range_t {0, m_IndicesSegment.range().size()};
111  }
112  else
113  {
114  core::igles::log->critical("index buffer was out of memory");
115  // todo error condition could not allocate segment
116  exit(1);
117  }
118 
119  m_IndicesBuffer->set({{(size_t)data->indices().data(),
120  m_IndicesSegment.range().begin,
121  data->indices().size() * sizeof(core::data::geometry_t::index_size_t)}});
122  }
123  else
124  {
125  auto& instr = instructions.emplace_back();
126  instr.size = data->indices().size() * sizeof(decltype(data->indices().at(0)));
127  instr.destination_offset = current_segment->first.range().begin + current_segment->second.begin;
128  instr.source_offset = (std::uintptr_t)(data->indices().data());
129 
130  m_IndicesSegment = current_segment->first;
131  m_IndicesSubRange = current_segment->second;
132  }
133 
134  m_Triangles = (m_IndicesSubRange.size() * sizeof(core::data::geometry_t::index_size_t)) / 3u;
135 
136  m_GeometryBuffer->set(instructions);
137  error = glGetError();
138 }

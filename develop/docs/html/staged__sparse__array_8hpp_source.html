468  auto erase_impl(chunk_t& chunk, index_t offset, index_t user_index)
469  {
470  auto orig_value = this->operator[](user_index);
471  auto reverse_index = chunk[offset];
472 
473  // chunk[offset] = std::numeric_limits<index_t>::max();
474  // figure out which stage it belonged to
475  auto what_stage = (reverse_index < m_StageStart[1]) ? 0 : (reverse_index < m_StageStart[2]) ? 1 : 2;
476  if(what_stage == 2) return;
477 
478  // we swap it out
479  for(auto i = what_stage; i < 2; ++i)
480  {
481  if(reverse_index != m_StageStart[i + 1] - 1)
482  {
483  std::iter_swap(std::next(std::begin(m_Reverse), reverse_index),
484  std::next(std::begin(m_Reverse), m_StageStart[i + 1] - 1));
485  std::iter_swap((T*)m_DenseData.data() + reverse_index,
486  (T*)m_DenseData.data() + (m_StageStart[i + 1] - 1));
487  chunk[offset] = m_StageStart[i + 1] - 1;
488  auto new_index = m_Reverse[reverse_index];
489  auto& new_chunk = chunk_for(new_index);
490  new_chunk[new_index] = reverse_index;
491  reverse_index = m_StageStart[i + 1] - 1;
492  }
493  m_StageStart[i + 1] -= 1;
494  }
495 
496 
497  // decrement that stage's size
498  m_StageSize[what_stage] -= 1;
499 
500  m_StageSize[2] += 1;
501  // if constexpr(std::is_same<value_t, int>())
502  // assert(orig_value == this->at(user_index,2,2));
503 
504  assert(chunk[offset] == reverse_index);
505  }

507  void grow()
508  {
509  auto capacity = m_Reverse.capacity() + 1;
510  if(m_DenseData.size() < capacity * sizeof(T))
511  {
512  auto new_capacity = std::max(capacity, m_DenseData.size() * 2 / sizeof(T));
513 
514  ::memory::raw_region reg(new_capacity * sizeof(T));
515 
516  if constexpr(std::is_trivially_copyable_v<T>)
517  {
518  std::memcpy(reg.data(), m_DenseData.data(), m_DenseData.size());
519  }
520  else
521  {
522  auto src = (T*)m_DenseData.data(), end = src + size(0, 2), dst = (T*)reg.data();
523 
524  if constexpr(std::is_move_constructible_v<T>)
525  std::uninitialized_move(src, end, dst);
526  else
527  std::uninitialized_copy(src, end, dst);
528 
529  if constexpr(!std::is_trivially_destructible_v<T>)
530  {
531  for(auto it = src; it != end; ++it)
532  {
533  it->~T();
534  }
535  }
536  }
537 
538  m_DenseData = std::move(reg);
539 
540  m_Reverse.reserve((m_DenseData.size() / sizeof(T)) - 1);
541 
542  assert_debug_break((m_Reverse.capacity() + 1) * sizeof(value_t) <= m_DenseData.size() &&
543  (m_Reverse.capacity() + 2) * sizeof(value_t) >= m_DenseData.size());
544  }

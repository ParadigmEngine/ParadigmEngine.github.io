177 void material_t::bind()
178 {
179  if(!m_Program) return;
180  glUseProgram(m_Program->id());
181  const auto& blend_states = m_Data->blend_states();
182  using namespace core::gfx::conversion;
183 
184 
185  for(auto i = 0; i < m_Textures.size(); ++i)
186  {
187  auto binding = m_Textures[i].first;
188 
189  if(binding == std::numeric_limits<uint32_t>::max()) continue;
190 
191  glActiveTexture(GL_TEXTURE0 + binding);
192  auto tex_id = m_Textures[i].second->id();
193  glBindTexture(GL_TEXTURE_2D, tex_id);
194  glBindSampler(binding, m_Samplers[i].second->id());
195  }
196 
197  for(auto i = 0; i < blend_states.size(); ++i)
198  {
199  glBlendEquationSeparatei(
200  i, to_gles(blend_states[i].color_blend_op()), to_gles(blend_states[i].alpha_blend_op()));
201  glBlendFuncSeparatei(i,
202  to_gles(blend_states[i].color_blend_src()),
203  to_gles(blend_states[i].color_blend_dst()),
204  to_gles(blend_states[i].alpha_blend_src()),
205  to_gles(blend_states[i].alpha_blend_dst()));
206  }
207  for(auto& buffer : m_Buffers)
208  {
209  if(buffer.offset == 0)
210  glBindBufferBase(GL_UNIFORM_BUFFER, buffer.slot, buffer.buffer->id());
211  else
212  glBindBufferRange(GL_UNIFORM_BUFFER, buffer.slot, buffer.buffer->id(), buffer.offset, buffer.size);
213  }
214 
215  switch(m_Data->cull_mode())
216  {
217  case core::gfx::cullmode::front:
218  glEnable(GL_CULL_FACE);
219  glCullFace(GL_FRONT);
220  break;
221  case core::gfx::cullmode::back:
222  glEnable(GL_CULL_FACE);
223  glCullFace(GL_BACK);
224  break;
225  case core::gfx::cullmode::all:
226  glEnable(GL_CULL_FACE);
227  glCullFace(GL_FRONT_AND_BACK);
228  break;
229  case core::gfx::cullmode::none:
230  glDisable(GL_CULL_FACE);
231  break;
232  }
233 
234  glDepthMask(m_Data->depth_write());
